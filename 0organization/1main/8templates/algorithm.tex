\begin{algorithm}[!htbp]%Adding float-specifier brackets, makes it behave as a float, otherwise not
    \caption{Euclid’s algorithm}%
    \label{euclid}%
    \begin{algorithmic}[1]\denkrAlgorithmFont% The number tells where the line numbering should start
        \Procedure{Euclid}{$a,b$} \Comment{The g.c.d. of a and b}
            \State $r\gets a \bmod b$
            \While{$r\not=0$} \Comment{We have the answer if r is 0}
                \State $a \gets b$
                \State $b \gets r$
                \State $r \gets a \bmod b$
            \EndWhile\label{euclidendwhile}
            \State \textbf{return} $b$\Comment{The gcd is b}
        \EndProcedure
    \end{algorithmic}%
\end{algorithm}%




\begin{algorithm_breakable}%
  \caption{Euclid’s algorithm}%
  \begin{algorithmic}[1]\denkrAlgorithmFont%
    \Procedure{Euclid}{$a,b$}\Comment{The g.c.d.\ of~$a$ and~$b$}
    \State $r \gets a \bmod b$
    \While{$r \neq 0$}\Comment{We have the answer if~$r$ is~$0$}
      \State $a \gets b$
      \State $b \gets r$
      \State $r \gets a \bmod b$
    \EndWhile
    \State \textbf{return} $b$\Comment{The g.c.d.\ is~$b$}
    \EndProcedure
  \end{algorithmic}%
\end{algorithm_breakable}%
